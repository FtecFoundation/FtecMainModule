Register new user

Register new user

This url is supposed create a new user
If user was created successfully returns response with status 200 and token
If username already exists returns 400 status with code 3
If email already exists returns 400 status with code 4

REQUEST Create user

POST

/registration
Headers
Content-Type application/json
application/json
BODY SCHEMA 
                          {
  "username": "login",
  "password": "Password123",
  "email": "example@email.com",
  "subscribeForEmail": true,
  "referrerId": 1
}
                        
                          {
  "$schema": "http://json-schema.org/draft-04/schema#",
  "type": "object",
  "properties": {
    "username": {
      "type": "string",
      "description": "@NotNull, @Size(min = 4, max = 20), @UniqueLogin User's login"
    },
    "password": {
      "type": "string",
      "description": "@NotNull, @Size(max = 20), @Pattern(regexp = ^(?=.*[0-9])(?=.*[a-z])(?=.*[A-Z])(?=\\S+$).{8,}$) User's password"
    },
    "email": {
      "type": "string",
      "description": "@NotNull, @UniqueEmail, @Size(max = 20) User's email"
    },
    "subscribeForEmail": {
      "type": "boolean",
      "description": "Is user want to be subscribed to our emails"
    },
    "referrerId": {
      "type": "number",
      "description": "Optional. If sent user will be counted as a referrer for `id` user"
    }
  }
}

RESPONSE

Headers
Content-Type application/json
200 application/json
BODY SCHEMA 
                        {
    "status": 0,
    "response": "123_token"
}
